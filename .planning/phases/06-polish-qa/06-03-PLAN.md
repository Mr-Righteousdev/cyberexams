```php
use Illuminate\Foundation\Testing\RefreshDatabase;

it('prevents student from accessing admin routes', function () {
    $student = User::factory()->create(['role' => 'student']);

    $adminRoutes = [
        '/admin',
        '/admin/exams',
        '/admin/students',
        '/admin/results',
        '/admin/monitor',
        '/admin/flagged',
    ];

    foreach ($adminRoutes as $route) {
        actingAs($student)
            ->get($route)
            ->assertStatus(403);
    }
});

it('allows admin to access admin routes', function () {
    $admin = User::factory()->create(['role' => 'admin']);

    actingAs($admin)
        ->get('/admin')
        ->assertStatus(200);
});

it('redirects unauthenticated users from protected routes', function () {
    $publicAdminRoutes = ['/admin/exams', '/admin/students'];

    foreach ($publicAdminRoutes as $route) {
        get($route)->assertRedirect('/login');
    }
});
```

```php
<?php

namespace Tests\Browser;

use App\Models\User;
use App\Models\Exam;
use Illuminate\Foundation\Testing\RefreshDatabase;
use Laravel\Dusk\Browser;
use Tests\DuskTestCase;

class SecuritySmokeTest extends DuskTestCase
{
    use RefreshDatabase;

    /** @test */
    public function student_cannot_access_admin_url_directly(): void
    {
        $student = User::factory()->create(['role' => 'student']);

        $this->browse(function (Browser $browser) use ($student) {
            $browser
                ->loginAs($student)
                ->visit('/admin')
                ->assertSee('Access Denied'); // or 403 page
        });
    }

    /** @test */
    public function exam_auto_submits_when_timer_expires(): void
    {
        $admin = User::factory()->create(['role' => 'admin']);
        $student = User::factory()->create(['role' => 'student', 'is_active' => true]);
        $exam = Exam::factory()->create([
            'is_published' => true,
            'duration_minutes' => 1, // 1 minute exam
        ]);

        // Start session
        $browser = new Browser;
        $browser
            ->loginAs($student)
            ->visit('/student/exam/' . $exam->id)
            ->click('@start-exam');

        // Wait for timer to reach zero (61 seconds for 1-minute exam + buffer)
        $browser->pause(62000);

        // Should be redirected to results
        $browser
            ->waitForLocation('/student/results', 10)
            ->assertPathIs('/student/results');
    }

    /** @test */
    public function network_blocked_page_shown_from_external_ip(): void
    {
        // This test requires IP spoofing or is skipped in local testing
        // Mark as @skip in local env
        $this->browse(function (Browser $browser) {
            $browser
                ->visit('/')
                ->assertStatus(200); // network blocked only for protected routes
        });
    }

    /** @test */
    public function no_way_to_access_admin_routes_via_ui_navigation(): void
    {
        $student = User::factory()->create(['role' => 'student']);

        $this->browse(function (Browser $browser) use ($student) {
            $browser
                ->loginAs($student)
                ->visit('/student')
                ->assertDontSee('/admin') // No admin links in student nav
                ->assertDontSee('@admin'); // No admin buttons
        });
    }
}
```
