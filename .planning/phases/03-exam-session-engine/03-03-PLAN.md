Create `app/Services/GradingService.php`:

```php
namespace App\Services;

use App\Models\ExamSession;
use App\Models\Answer;
use App\Models\Question;

class GradingService
{
    public static function grade(ExamSession $session): void
    {
        // Already graded
        if ($session->score !== null) {
            return;
        }

        $exam = $session->exam;
        $answers = $session->answers()->with(['question', 'selectedOption'])->get();
        $totalMarks = 0;
        $awardedMarks = 0;
        $answeredCount = 0;

        foreach ($answers as $answer) {
            $question = $answer->question;
            $totalMarks += $question->marks;

            $correctOption = $question->options()->where('is_correct', true)->first();

            if ($question->type === 'mcq' || $question->type === 'true_false') {
                $isCorrect = $answer->selected_option_id !== null
                    && $correctOption !== null
                    && $answer->selected_option_id === $correctOption->id;

                $answer->update([
                    'is_correct' => $isCorrect,
                    'marks_awarded' => $isCorrect ? $question->marks : 0,
                ]);

                if ($isCorrect) {
                    $awardedMarks += $question->marks;
                }
                $answeredCount++;
            } elseif ($question->type === 'short_answer' || $question->type === 'code_snippet') {
                // Leave null — manual grading
                $answeredCount++;
            }
        }

        $percentage = $totalMarks > 0 ? round(($awardedMarks / $totalMarks) * 100, 2) : 0;
        $passed = $exam->passing_marks !== null
            ? ($awardedMarks / $totalMarks * 100) >= ($exam->passing_marks / $totalMarks * 100)
            : null; // no passing threshold set

        $session->update([
            'score' => $awardedMarks,
            'total_marks' => $totalMarks,
            'percentage' => $percentage,
            'passed' => $passed,
        ]);
    }
}
```

Note: `total_marks` column is on ExamSession but not in the existing fillable. Add it to ExamSession model's fillable and casts:

```php

    ...existing...,
    'total_marks',
];

protected $casts = [
    ...existing...,
    'total_marks' => 'integer',
];
```
